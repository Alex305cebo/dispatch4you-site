### 3. JavaScript функции

```javascript
// Обновлена функция updateProgress
function updateProgress(audioId, currentTime, duration) {
    const container = document.querySelector(`[onclick*="${audioId}"]`).closest('.audio-container');
    const progressRing = container.querySelector('.progress-ring-fill');
    const mobileProgressFill = container.querySelector('.mobile-progress-fill');
    const durationDisplay = container.querySelector('.audio-duration');

    const progress = currentTime / duration;
    
    // Update ring progress (desktop)
    if (progressRing) {
        progressRing.style.strokeDashoffset = offset;
    }

    // Update mobile progress bar
    if (mobileProgressFill) {
        mobileProgressFill.style.width = `${progress * 100}%`;
    }

    durationDisplay.textContent = `${formatTime(currentTime)} / ${formatTime(duration)}`;
}

// Новая функция для перемотки на мобильных
function seekAudioMobile(event, element, audioId) {
    event.stopPropagation();

    const audio = document.getElementById(audioId);
    if (!audio || audio.paused) {
        return; // Only allow seeking when playing
    }

    const rect = element.getBoundingClientRect();
    const clickX = event.clientX - rect.left;
    const progress = clickX / rect.width;
    const duration = audioDurations[audioId];
    const newTime = progress * duration;

    // Update progress immediately
    updateProgress(audioId, newTime, duration);

    // Seek audio (для audio1 с реальным файлом)
    if (audioId === 'audio1' && audio.duration) {
        audio.currentTime = newTime;
    }
}
```
